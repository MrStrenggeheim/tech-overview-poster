\newcommand{\cellmargin}{2pt} 

% Logo with fallback
\newcommand{\logo}[1]{%
  \IfFileExists{logos/#1.pdf}{%
    \includegraphics[height=0.6cm]{logos/#1.pdf}%
  }{%
    \IfFileExists{logos/#1.png}{%
      \includegraphics[height=0.6cm]{logos/#1.png}%
    }{%
      \textbf{?}%
    }%
  }%
} 

% Small inline logo (for cells)
\newcommand{\smalllogo}[1]{%
  \IfFileExists{logos/#1.pdf}{%
    \includegraphics[height=0.4cm]{logos/#1.pdf}%
  }{%
    \IfFileExists{logos/#1.png}{%
      \includegraphics[height=0.4cm]{logos/#1.png}%
    }{%
    }%
  }%
}

% Product cell: logo + name (bold) + description (subtle)
\newcommand{\product}[3]{%
  \begin{minipage}{\linewidth}%
    \centering%
    \vspace{\cellmargin}%
    \smalllogo{#1}\\[-4pt]% Tightened distance
    {\fontsize{9}{10}\selectfont\textbf{#2}}\\[-3pt]% Tightened distance
    {\fontsize{7}{8}\selectfont\color{subtletext}#3}%
    \vspace{\cellmargin}%
\end{minipage}%
}

% Product cell with multiple logos
\newcommand{\productmulti}[5]{%
  \begin{minipage}{\linewidth}%
    \centering%
    \vspace{\cellmargin}%
    \smalllogo{#1}\hspace{2pt}\smalllogo{#2}\hspace{2pt}\smalllogo{#3}\\[-4pt]%
    {\fontsize{9}{10}\selectfont\textbf{#4}}\\[-3pt]%
    {\fontsize{7}{8}\selectfont\color{subtletext}#5}%
    \vspace{\cellmargin}%
\end{minipage}%
}

% Empty cell placeholder
\newcommand{\emptycell}{%
\begin{minipage}{\linewidth}%
    \centering%
    \vspace{\cellmargin}%
    \color{subtletext}—%
    \vspace{\cellmargin}%
\end{minipage}%
}

% Category label (row header)
\newcommand{\category}[1]{%
\begin{minipage}{\linewidth}%
    \vspace{\cellmargin}%
    \raggedright%
    {\fontsize{9}{10}\selectfont\textbf{#1}}%
    \vspace{\cellmargin}%
  \end{minipage}%
}

% Company label (column header)
\newcommand{\company}[2]{%
  \begin{minipage}{\linewidth}%
    \centering%
    \vspace{\cellmargin}%
    \ifx&#1&\else\logo{#1}\\[0pt]\fi%
    {\fontsize{9}{10}\selectfont\textbf{#2}}%
    \vspace{\cellmargin}%
  \end{minipage}%
}
